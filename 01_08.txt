      function getData()
      {
        xmldoc= document.all("firstXML").XMLDocument;

        nodeDoc = xmldoc.documentElement;
        nodeGreeting = nodeDoc.firstChild;

        outputMessage = "Greeting: " +
                nodeGreeting.firstChild.nodeValue;
        message.innerHTML=outputMessage;
      }
